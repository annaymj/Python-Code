def percent_id(aln_seq1,aln_seq2):
    i_score=0
    t=len(aln_seq1)
    for i in range(len(aln_seq1)):        #len(aln_seq1)=len(aln_seq2)
        if aln_seq1[i]==aln_seq2[i]:
            i_score+=1
    print (float(i_score)/float(t))*100   #This function is based on all previous functions, I have attached the whole .py file as well.
